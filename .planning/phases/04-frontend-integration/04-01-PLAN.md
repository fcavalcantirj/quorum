3. Create `web/lib/api.ts` — typed fetch wrapper for client AND server components:
   ```typescript
   // For "use client" components — sends session cookie cross-origin
   const API_URL = process.env.NEXT_PUBLIC_API_URL

   export async function apiFetch<T>(path: string, options?: RequestInit): Promise<T> {
     const res = await fetch(`${API_URL}${path}`, {
       credentials: "include",
       headers: { "Content-Type": "application/json", ...options?.headers },
       ...options,
     })
     if (!res.ok) {
       const body = await res.json().catch(() => ({ error: `API error ${res.status}` }))
       throw new Error(body.error || `API error ${res.status}`)
     }
     return res.json()
   }

   // For server components — uses server-only API_URL (no NEXT_PUBLIC_)
   export async function serverFetch<T>(path: string, options?: RequestInit): Promise<T> {
     const res = await fetch(`${process.env.API_URL}${path}`, {
       cache: "no-store",
       headers: { "Content-Type": "application/json", ...options?.headers },
       ...options,
     })
     if (!res.ok) {
       const body = await res.json().catch(() => ({ error: `API error ${res.status}` }))
       throw new Error(body.error || `API error ${res.status}`)
     }
     return res.json()
   }
   ```

4. Create `web/lib/session.ts` — server-only session verification:
   ```typescript
   import "server-only"
   import { cookies } from "next/headers"
   import { jwtVerify } from "jose"
   import type { SessionPayload } from "./types"

   const secret = new TextEncoder().encode(process.env.SESSION_SECRET)

   export async function getSession(): Promise<SessionPayload | null> {
     const cookieStore = await cookies()
     const token = cookieStore.get("session")?.value
     if (!token) return null
     try {
       const { payload } = await jwtVerify(token, secret, { algorithms: ["HS256"] })
       return payload as unknown as SessionPayload
     } catch {
       return null
     }
   }
   ```

5. Create `web/lib/dal.ts` — data access layer:

   ```typescript
   import "server-only"
   import { getSession } from "./session"
   import { redirect } from "next/navigation"

   export async function verifySession() {
     const session = await getSession()
     if (!session) redirect("/login")
     return session
   }
   ```

6. Create `web/middleware.ts`:
   ```typescript
   import { NextRequest, NextResponse } from "next/server"
   import { jwtVerify } from "jose"

   const secret = new TextEncoder().encode(process.env.SESSION_SECRET)
   const protectedRoutes = ["/dashboard"]

   export default async function middleware(req: NextRequest) {
     const path = req.nextUrl.pathname
     if (!protectedRoutes.some(r => path.startsWith(r))) return NextResponse.next()

     const cookie = req.cookies.get("session")?.value
     try {
       await jwtVerify(cookie ?? "", secret, { algorithms: ["HS256"] })
       return NextResponse.next()
     } catch {
       return NextResponse.redirect(new URL("/login", req.nextUrl))
     }
   }

   export const config = {
     matcher: ["/((?!api|_next/static|_next/image|.*\\.png$).*)"],
   }
   ```

   ```tsx
   // web/app/(marketing)/_components/ActivityWidget.tsx
   import Link from "next/link"
   import { Button } from "@/components/ui/button"
   import { Skeleton } from "@/components/ui/skeleton"
   import { serverFetch } from "@/lib/api"
   import type { Stats } from "@/lib/types"

   export async function ActivityWidget() {
     let stats: Stats | null = null
     try {
       stats = await serverFetch<Stats>("/stats")
     } catch {
       // API unreachable — render skeletons
     }

     const counters = [
       { value: stats?.activeRooms, label: "Public Rooms" },
       { value: stats?.agentsOnline, label: "Agents Online" },
       { value: stats?.messagesRelayed, label: "Messages Relayed" },
     ]

     return (
       <section className="w-full bg-card border border-border rounded-lg py-12 px-8">
         <h2 className="text-center text-sm font-semibold uppercase tracking-wider text-muted-foreground mb-8">
           Quorum Network — Live
         </h2>
         <div className="grid grid-cols-1 md:grid-cols-3 gap-6 text-center">
           {counters.map((c) => (
             <div key={c.label} className="flex flex-col items-center gap-1">
               {c.value != null ? (
                 <span className="text-4xl font-semibold text-primary tabular-nums">
                   {c.value.toLocaleString()}
                 </span>
               ) : (
                 <Skeleton className="h-10 w-20" />
               )}
               <span className="text-xs text-muted-foreground">{c.label}</span>
             </div>
           ))}
         </div>
         <div className="flex justify-center mt-8">
           <Button asChild>
             <Link href="/explore">See all public rooms</Link>
           </Button>
         </div>
       </section>
     )
   }
   ```
