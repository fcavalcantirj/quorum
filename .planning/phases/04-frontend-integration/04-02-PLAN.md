1. Create `web/hooks/use-sse.ts` — a generic reusable SSE hook:
   ```typescript
   "use client"
   import { useEffect, useRef, useState } from "react"

   interface UseSSEOptions {
     onMessage?: (event: MessageEvent) => void
     onEvent?: Record<string, (event: MessageEvent) => void>
   }

   export function useSSE(url: string | null, options: UseSSEOptions = {}) {
     const [connected, setConnected] = useState(false)
     const [error, setError] = useState(false)
     const esRef = useRef<EventSource | null>(null)

     useEffect(() => {
       if (!url) return

       const es = new EventSource(url, { withCredentials: true })
       esRef.current = es

       es.onopen = () => { setConnected(true); setError(false) }
       es.onerror = () => { setConnected(false); setError(true) }

       if (options.onMessage) {
         es.onmessage = options.onMessage
       }
       if (options.onEvent) {
         for (const [eventName, handler] of Object.entries(options.onEvent)) {
           es.addEventListener(eventName, handler)
         }
       }

       return () => { es.close(); setConnected(false) }
     }, [url])

     return { connected, error }
   }
   ```

   "use client"
   import { useState } from "react"
   import { useSSE } from "@/hooks/use-sse"
   import type { Stats } from "@/lib/types"

   interface StatsBarProps {
     initialStats: Stats
   }

   export function StatsBar({ initialStats }: StatsBarProps) {
     const [stats, setStats] = useState<Stats>(initialStats)
     const apiUrl = process.env.NEXT_PUBLIC_API_URL

     useSSE(apiUrl ? `${apiUrl}/stats/stream` : null, {
       onEvent: {
         stats_update: (event) => {
           try {
             const updated = JSON.parse(event.data) as Stats
             setStats(updated)
           } catch { /* ignore malformed events */ }
         },
       },
     })

     const counters = [
       { value: stats.activeRooms, label: "Public Rooms" },
       { value: stats.agentsOnline, label: "Agents Online" },
       { value: stats.messagesRelayed, label: "Messages Relayed" },
     ]

     return (
       <section className="w-full bg-card border border-border rounded-lg py-8 px-6">
         <h2 className="text-center text-sm font-semibold uppercase tracking-wider text-muted-foreground mb-6">
           Quorum Network — Live

           {counters.map((c) => (
             <div key={c.label} className="flex flex-col items-center gap-1">
               <span className="text-4xl font-semibold text-primary tabular-nums">

               </span>
               <span className="text-xs text-muted-foreground">{c.label}</span>
             </div>
           ))}

3. Create `web/app/explore/_components/RoomGrid.tsx` — client component managing search state and rendering:
   ```tsx
   "use client"
   import { useMemo, useState } from "react"
   import type { Room } from "@/lib/types"
   import { RoomCard } from "./RoomCard"
   import { SearchFilter } from "./SearchFilter"

   interface RoomGridProps {
     initialRooms: Room[]
   }

   export function RoomGrid({ initialRooms }: RoomGridProps) {
     const [query, setQuery] = useState("")
     const [selectedTags, setSelectedTags] = useState<string[]>([])

     const availableTags = useMemo(() => {
       const tagSet = new Set<string>()
       initialRooms.forEach((r) => r.tags.forEach((t) => tagSet.add(t)))
       return Array.from(tagSet).sort()
     }, [initialRooms])

     const filtered = useMemo(() => {
       return initialRooms.filter((room) => {
         const matchesQuery =
           !query || room.name.toLowerCase().includes(query.toLowerCase())
         const matchesTags =
           selectedTags.length === 0 ||
           selectedTags.some((t) => room.tags.includes(t))
         return matchesQuery && matchesTags
       })
     }, [initialRooms, query, selectedTags])

     const handleTagToggle = (tag: string) => {
       setSelectedTags((prev) =>
         prev.includes(tag) ? prev.filter((t) => t !== tag) : [...prev, tag]
       )
     }

     return (
       <div className="space-y-8">
         <SearchFilter
           query={query}
           onQueryChange={setQuery}
           selectedTags={selectedTags}
           onTagToggle={handleTagToggle}
           availableTags={availableTags}
         />

         {filtered.length > 0 ? (
           <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
             {filtered.map((room) => (
               <RoomCard key={room.id} room={room} />
             ))}
           </div>
         ) : initialRooms.length === 0 ? (
           <div className="text-center py-16 space-y-2">
             <h3 className="text-lg font-semibold">No public rooms yet</h3>
             <p className="text-muted-foreground text-sm">
               Be the first to create one. Rooms are free and open to any agent.
             </p>
           </div>
         ) : (
           <div className="text-center py-16 space-y-2">
             <h3 className="text-lg font-semibold">No rooms match that search</h3>
             <p className="text-muted-foreground text-sm">
               Try a different name or tag, or create a new room for your use case.
             </p>
           </div>
         )}
       </div>
     )
   }
   ```
