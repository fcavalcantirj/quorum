    import "github.com/google/uuid"

    // RoomID is a type-safe wrapper around uuid.UUID to prevent cross-room contamination bugs.
    type RoomID uuid.UUID

    func NewRoomID(id uuid.UUID) RoomID { return RoomID(id) }
    func (r RoomID) UUID() uuid.UUID   { return uuid.UUID(r) }
    func (r RoomID) String() string    { return uuid.UUID(r).String() }

    func ParseRoomID(s string) (RoomID, error) {
        id, err := uuid.Parse(s)
        if err != nil {
            return RoomID{}, err
        }
        return RoomID(id), nil
    }

    **hub.go** — Per-room goroutine hub per D-08 and research Pattern 1:
    ```go
    package hub

    import (
        "context"
        "log/slog"
        "time"
    )

    type subscribeCmd struct {
        agentName string
        card      *a2a.AgentCard
        ch        chan<- RoomEvent
        resp      chan error
    }

    type unsubscribeCmd struct {
        agentName string
    }

    type broadcastCmd struct {
        event RoomEvent
    }

    type RoomHub struct {
        ID          RoomID
        subscribe   chan subscribeCmd
        unsubscribe chan unsubscribeCmd
        broadcast   chan broadcastCmd
        events      chan RoomEvent  // consumed by Phase 3 SSE; buffered channel (size 256)
        done        chan struct{}
        logger      *slog.Logger
    }

    func NewRoomHub(id RoomID, registry *PresenceRegistry, logger *slog.Logger) *RoomHub { ... }

    // Run starts the hub goroutine. It owns all mutable subscriber state.
    // Exits when ctx is canceled.
    func (h *RoomHub) Run(ctx context.Context, registry *PresenceRegistry) {
        subscribers := make(map[string]chan<- RoomEvent)
        for {
            select {
            case cmd := <-h.subscribe:
                subscribers[cmd.agentName] = cmd.ch
                registry.Add(h.ID, cmd.agentName, cmd.card)
                // Emit agent_joined event to all existing subscribers
                evt := RoomEvent{Type: EventAgentJoined, RoomID: h.ID, AgentName: cmd.agentName, Payload: cmd.card, Timestamp: time.Now()}
                for name, ch := range subscribers {
                    if name != cmd.agentName {
                        select {
                        case ch <- evt:
                        default: // slow consumer — drop event
                        }
                    }
                }
                // Also emit to the events channel for Phase 3 SSE
                select {
                case h.events <- evt:
                default:
                }
                cmd.resp <- nil
            case cmd := <-h.unsubscribe:
                if ch, ok := subscribers[cmd.agentName]; ok {
                    delete(subscribers, cmd.agentName)
                    close(ch)
                }
                registry.Remove(h.ID, cmd.agentName)
                evt := RoomEvent{Type: EventAgentLeft, RoomID: h.ID, AgentName: cmd.agentName, Timestamp: time.Now()}
                for _, ch := range subscribers {
                    select {
                    case ch <- evt:
                    default:
                    }
                }
                select {
                case h.events <- evt:
                default:
                }
            case cmd := <-h.broadcast:
                for _, ch := range subscribers {
                    select {
                    case ch <- cmd.event:
                    default: // slow consumer — drop
                    }
                }
            case <-ctx.Done():
                // Cleanup: close all subscriber channels
                for _, ch := range subscribers {
                    close(ch)
                }
                close(h.done)
                return
            }
        }
    }

    // Subscribe adds an agent to the room. Returns a channel to receive events.
    func (h *RoomHub) Subscribe(agentName string, card *a2a.AgentCard) (<-chan RoomEvent, error) {
        ch := make(chan RoomEvent, 64) // buffered per research: avoid hub starvation
        resp := make(chan error, 1)
        h.subscribe <- subscribeCmd{agentName: agentName, card: card, ch: ch, resp: resp}
        return ch, <-resp
    }

    // Unsubscribe removes an agent from the room.
    func (h *RoomHub) Unsubscribe(agentName string) {
        h.unsubscribe <- unsubscribeCmd{agentName: agentName}
    }

    // Broadcast sends an event to all subscribers in the room.
    func (h *RoomHub) Broadcast(event RoomEvent) {
        h.broadcast <- broadcastCmd{event: event}
    }
    ```

       -- name: UpsertAgentPresence :one
       INSERT INTO agent_presence (room_id, agent_name, card_json, ttl_seconds)
       VALUES ($1, $2, $3, $4)
       ON CONFLICT (room_id, agent_name)
       DO UPDATE SET card_json = EXCLUDED.card_json, last_seen = NOW(), ttl_seconds = EXCLUDED.ttl_seconds

       UPDATE agent_presence SET last_seen = NOW() WHERE room_id = $1 AND agent_name = $2;

       DELETE FROM agent_presence

       SELECT ap.* FROM agent_presence ap
       JOIN rooms r ON r.id = ap.room_id

       AND ap.last_seen > NOW() - (ap.ttl_seconds || ' seconds')::interval
       ORDER BY ap.last_seen DESC;
